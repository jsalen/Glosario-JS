Glosario de JavaScript

alert()

Muestra una alerta en el navegador.

alert("Hello! I am an alert box!!");

appendChild()

Añade un nodo como último hijo de un nodo.

var node = document.getElementById("myList2").lastChild;
document.getElementById("myList1").appendChild(node);

charAt()

Retorna el carácter en el index especificado.

var str = "HELLO WORLD";
var res = str.charAt(0); // Salida: "H"

concat()

Une dos o más arrays.

var hege = ["Cecilie", "Lone"];
var stale = ["Emil", "Tobias", "Linus"];
var kai = ["Robin"];
var children = hege.concat(stale, kai);

confirm()

Muestra una caja de dialogo con un mensaje, un botón de aceptar y un botón de cancelar.

confirm("Estás de acuerdo?");

console

Referencia al objeto 'console', el más usado es log.

console.log("Mensaje mostrado en consola");

createTextNode()

Crea un nodo de texto.

var t = document.createTextNode("This is a paragraph.");

debugger

Detiene la ejecución de Javascript, e invoca (si es posible) la función de debugging.

var x = 15 * 5;
debugger;
document.getElementbyId("demo").innerHTML = x;

do ... while

Crea un loop que ejecuta un bloque de código antes de chequear si la condición es cierta, luego repite el código hasta que la condición sea cierta.

var i = 0;
do {
console.log(i);
i++;
}
while (i < 5);

entries()

Retorna los keys de un array.

var fruits = ["Banana", "Orange", "Apple", "Mango"];
var f = fruits.entries();

every()

Evalua si todos los elementos de un array pasa una prueba.

var ages = [32, 33, 16, 40];

function checkAdult(age) {
return age >= 18;
}

function myFunction() {
document.getElementById("demo").innerHTML = ages.every(checkAdult);
}

filter()

Crea un array con los elementos que pasen una prueba

var ages = [32, 33, 16, 40];

function checkAdult(age) {
return age >= 18;
}

function myFunction() {
document.getElementById("demo").innerHTML = ages.filter(checkAdult);
}

function

Declara una función

function miFuncion () {
console.log("Una funcion en JS");
}

import

Importa una función exportada desde un módulo externo

import react from 'React';

indexOf()

Busca en el array por un elemento y retorna su index

var fruits = ["Banana", "Orange", "Apple", "Mango"];
var a = fruits.indexOf("Apple");

innerHTML

Establece o retorna el contenido de un elemento

document.getElementById("demo").innerHTML = "Nuevo párrafo!";

length

Retorna la cantidad de elementos en un array

var fruits = ["Banana", "Orange", "Apple", "Mango"];
fruits.length;

let

Declara una variable de alcance de bloque

let name = 'George';

var

Declara una variable de alcance global

var texto = 'Algo de texto';

const

Establece el valor de una constante

const SECONDS_PER_HOUR = 3600;
